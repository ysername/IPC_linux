143 /*
144 ** HMI_SCHEDULING_RATE_MS - This value of this macro must be set to indicate the periodic
145 ** scheduling rate of the "HMISS_PeriodicTask" function.
146 */
147 #define HMI_SCHEDULING_RATE_MS (17u) //58.8 FPS ~60FPS
148 
149 /*
150 ** HMI_ENABLE_CHRONOMETRICS - Macro that controls whether HMI chronometrics are captured
151 ** Set to 0 to DISable chronometrics
152 ** Set to non-0 to ENable chronometrics
153 */
154 #define HMI_ENABLE_CHRONOMETRICS (0u)
155 
156 
157 /*
158 ** HMI_USE_VS_STUBS - Defines if Visual State stub functions should be included.
159 ** 0 = Do NOT include stubs
160 ** non-0 = Include stubs (for compiling/linking and testing without HMI model).
161 */
162 #define HMI_USE_VS_STUBS (0)
163 
164 /*
165 ** FRAME_RATE_INSTANT_SAMPLES - The number of samples used to compute instanteous
166 ** frame rate. A good value for this parameter is 1000/HMI_SCHEDULING_RATE_MS
167 ** (instanteous frame rate compute over samples taken in the last second).
168 ** The value range for this parameter is 1..1000.
169 */
170 #define FRAME_RATE_INSTANT_SAMPLES (1000/HMI_SCHEDULING_RATE_MS)
171 
172 
173 /*
174 ** HMI_ENABLE_FRAME_RATE_SUPPORT - Macro that controls whether HMI frame rate computation is supported.
175 ** Set to 0 to disable frame rate support
176 ** Set to non-0 to enable frame rate support
177 */
178 #define HMI_ENABLE_FRAME_RATE_SUPPORT (0)
